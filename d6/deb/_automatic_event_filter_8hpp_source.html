81 inline void
82 AutomaticEventFilter::updateForNewEvent( LhefEvent const& eventSource )
83 /* this clears filteredLines then goes through eventSource & records in
84 * filteredLines all pointers to ParticleLines that pass the filters.
85 * this is automatically called by the LhefParser that created this
86 * instance when it reads in a new LHEF event, so it shouldn't have to be
87 * called by anything else.
88 */
89 {
90 filteredLines.clear();
91 for( int whichLine( eventSource.getNumberOfParticles() );
92 0 < whichLine;
93 --whichLine )
94 {
95 if( passesAllFilters( eventSource[ whichLine ] ) )
96 {
97 filteredLines.push_back( &(eventSource[ whichLine ]) );
98 }
99 }
100 }

102 inline bool
103 AutomaticEventFilter::passesAllFilters( ParticleLine const& lineToCheck )
104 /* this clears filteredLines then goes through eventSource & records in
105 * filteredLines all pointers to ParticleLines that pass the filters.
106 * this is automatically called by the LhefParser that created this
107 * instance when it reads in a new LHEF event, so it shouldn't have to be
108 * called by anything else.
109 */
110 {
111 rulesToPass = numberOfRules;
112 while( ( 0 < rulesToPass )
113 &&
114 ( (*(filterRules[ numberOfRules
115 - rulesToPass ]))( lineToCheck ) ) )
116 {
117 --rulesToPass;
118 }
119 return( 0 == rulesToPass );
120 }
