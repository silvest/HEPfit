82void MonteCarloEngine::Initialize(StandardModel* Mod_i)
83{
84 Mod = Mod_i;
85 int k = 0, kweight = 0;
86
87 for (boost::ptr_vector<Observable>::iterator it = Obs_ALL.begin(); it < Obs_ALL.end(); it++) {
88 if (!it->isTMCMC()) {
89 k++;
90 if (it->getDistr().compare("noweight") != 0) kweight++;
91 if (it->isWriteChain()) kchainedObs++;
92 }
93 thMin[it->getName()] = std::numeric_limits<double>::max();
94 thMax[it->getName()] = -std::numeric_limits<double>::max();
95 }
96 for (std::vector<Observable2D>::iterator it = Obs2D_ALL.begin(); it < Obs2D_ALL.end(); it++) {
97 if ((it->getDistr()).compare("file") == 0) {
98 if (!it->isTMCMC())
99 throw std::runtime_error("ERROR: cannot handle noMCMC for Observable2D file yet!");
100 } else if (it->getDistr().compare("weight") == 0)
101 throw std::runtime_error("ERROR: do not use Observable2D for analytic 2D weights!");
102 }
103 for (std::vector<CorrelatedGaussianObservables>::iterator it1 = CGO.begin(); it1 != CGO.end(); ++it1) {
104 std::vector<Observable> ObsV(it1->getObs());
105 for (std::vector<Observable>::iterator it = ObsV.begin(); it != ObsV.end(); ++it) {
106 if ((it->getDistr()).compare("file") == 0)
107 throw std::runtime_error("Cannot use file in CorrelatedGaussianObservables!");
108 if (!(it->isTMCMC())) {
109 k++;
110 if (it->getDistr().compare("noweight") != 0)
111 throw std::runtime_error("Cannot use weight in CorrelatedGaussianObservables!");
112 }
113 thMin[it->getName()] = std::numeric_limits<double>::max();
114 thMax[it->getName()] = -std::numeric_limits<double>::max();
115 }
116 if (it1->isPrediction()) {
117 CorrelationMap[it1->getName()] = new TPrincipal(it1->getObs().size(), "N");
118 }
119 }
120 kmax = k;
121 kwmax = kweight;
122
123 unknownParameters = Mod->getUnknownParameters();
124 DefineParameters();
125};

212void MonteCarloEngine::DefineParameters() {
213 // Add parameters to your model here.
214 // You can then use them in the methods below by calling the
215 // parameters.at(i) or parameters[i], where i is the index
216 // of the parameter. The indices increase from 0 according to the
217 // order of adding the parameters.
218 if (rank == 0) std::cout << "\nParameters varied in this run:" << std::endl;
219 unsigned int k = 0;
220 for (std::vector<ModelParameter>::const_iterator it = ModPars.begin();
221 it < ModPars.end(); it++) {
222 if (std::find(unknownParameters.begin(), unknownParameters.end(), it->getname()) == unknownParameters.end()) {
223 if (it->geterrf() == 0. && it->geterrg() == 0.)
224 continue;
225
226 AddParameter(it->getname().c_str(), it->getmin(), it->getmax());
227 if (rank == 0) std::cout << k << ": " << it->getname() << ", ";
228
229 if (it->IsCorrelated()) {
230 for (unsigned int i = 0; i < CGP.size(); i++) {
231 if (CGP[i].getName().compare(it->getCgp_name()) == 0) {
232 std::string index = it->getname().substr(CGP[i].getName().size());
233 long int lindex = strtol(index.c_str(), NULL, 10);
234 if (lindex > 0)
235 DPars[CGP[i].getPar(lindex - 1).getname()] = 0.;
236 else {
237 std::stringstream out;
238 out << it->getname();
239 throw std::runtime_error("MonteCarloEngine::DefineParameters(): " + out.str() + "seems to be part of a CorrelatedGaussianParameters object, but I couldn't find the corresponding object");
240 }
241 }
242 }
243 } else
244 DPars[it->getname()] = 0.;
245 if (it->geterrf() == 0.) GetParameter(k).SetPrior(std::make_shared<BCGaussianPrior>(it->getave(), it->geterrg())); //SetPriorGauss(k, it->getave(), it->geterrg());
246 else if (it->geterrg() == 0.) GetParameter(k).SetPriorConstant(); //SetPriorConstant(k);
247 else {
248 GetParameter(k).SetPrior(std::make_shared<BCCombinedPrior>(it->getave(), it->geterrg(), it->geterrf())); //SetPrior(k, combined);
249 }
250 k++;
251 }
252
253 }
254 if (unknownParameters.size() > 0 && rank == 0) {
255 std::cout << "\n" << std::endl;
256 for (std::vector<std::string>::iterator it = unknownParameters.begin(); it != unknownParameters.end(); it++)
257 std::cout << "WARNING: unknown parameter " << *it << " not added to MCMC" << std::endl;
258 }

261void MonteCarloEngine::setDParsFromParameters(const std::vector<double>& parameters,
262 std::map<std::string,double>& DPars_i)
263{
264 std::map<std::string, std::vector<double> > cgpmap;
265
266 unsigned int k = 0;
267 for (std::vector<ModelParameter>::const_iterator it = ModPars.begin(); it != ModPars.end(); it++){
268 if(it->IsFixed())
269 continue;
270 if (std::find(unknownParameters.begin(), unknownParameters.end(), it->getname()) != unknownParameters.end())
271 continue;
272 if(it->getname().compare(GetParameter(k).GetName()) != 0)
273 {
274 std::stringstream out;
275 out << it->getname();
276 throw std::runtime_error("MonteCarloEngine::setDParsFromParameters(): " + out.str() + "is sitting at the wrong position in the BAT parameters vector");
277 }
278 if (it->IsCorrelated()) {
279 std::string index = it->getname().substr(it->getCgp_name().size());
280 unsigned long int lindex = strtol(index.c_str(),NULL,10);
281 if (lindex - 1 == cgpmap[it->getCgp_name()].size())
282 cgpmap[it->getCgp_name()].push_back(parameters[k]);
283 else {
284 std::stringstream out;
285 out << it->getname() << " " << lindex;
286 throw std::runtime_error("MonteCarloEngine::setDParsFromParameters(): " + out.str() + "seems to be a CorrelatedGaussianParameters object but the corresponding parameters are missing or not in the right order");
287 }
288
289 } else
290 DPars_i[it->getname()] = parameters[k];
291 k++;
292 }
293
294 for (unsigned int j = 0; j < CGP.size(); j++) {
295 std::vector<double> current = cgpmap.at(CGP[j].getName());
296 if (current.size() != CGP[j].getPars().size()) {
297 std::stringstream out;
298 out << CGP[j].getName();
299 throw std::runtime_error("MonteCarloEngine::setDParsFromParameters(): " + out.str() + " appears to be represented in cgpmap with a wrong size");
300 }
301
302 std::vector<double> porig = CGP[j].getOrigParsValue(current);
303
304 for(unsigned int l = 0; l < porig.size(); l++) {
305 DPars_i[CGP[j].getPar(l).getname()] = porig[l];
306 }
307 }
308}

312double MonteCarloEngine::LogLikelihood(const std::vector<double>& parameters) {
313 // This methods returns the logarithm of the conditional probability
314 // p(data|parameters). This is where you have to define your model.
315
316 double logprob = 0.;
317
318 setDParsFromParameters(parameters, DPars);
319
320 // if update false set probability equal zero
321 if (!Mod->Update(DPars)) {
322#ifdef _MCDEBUG
323 std::cout << "event discarded" << std::endl;
324
325 /* Debug */
326 //for (int k = 0; k < parameters.size(); k++)
327 // std::cout << " " << GetParameter(k)->GetName() << " = "
328 // << DPars[GetParameter(k)->GetName()] << std::endl;
329#endif
330 NumOfDiscardedEvents++;
331 return (log(0.));
332 }
333#ifdef _MCDEBUG
334 //std::cout << "event used in MC" << std::endl;
335#endif
336
337 for (boost::ptr_vector<Observable>::iterator it = Obs_ALL.begin(); it != Obs_ALL.end(); it++) {
338 if (it->isTMCMC()) logprob += it->computeWeight();
339 }
340
341 for (std::vector<Observable2D>::iterator it = Obs2D_ALL.begin(); it != Obs2D_ALL.end(); it++) {
342 if (it->isTMCMC()) logprob += it->computeWeight();
343 }
344
345 for (std::vector<CorrelatedGaussianObservables>::iterator it = CGO.begin(); it < CGO.end(); it++) {
346 if(!(it->isPrediction())) logprob += it->computeWeight();
347 }
348 if (!std::isfinite(logprob) || !Mod->isQCDsuccess() || !Mod->isSMSuccess()) {
349 NumOfDiscardedEvents++;
350#ifdef _MCDEBUG
351// std::cout << "Event discarded since logprob evaluated to: " << logprob << std::endl ;
352#endif
353 return (log(0.));
354 }
355 NumOfUsedEvents++;
356 return logprob;
357}

359void MonteCarloEngine::MCMCUserIterationInterface() {
360#ifdef _MPI
361 unsigned mychain = 0;
362 int iproc = 0;
363 unsigned npars = GetNParameters();
364 int buffsize = npars + 1;
365 int index_chain[procnum];
366 double *recvbuff = new double[buffsize];
367 std::vector<double> pars;
368 double **buff;
369
370 buff = new double*[procnum];
371 int obsbuffsize = Obs_ALL.size() + 2 * Obs2D_ALL.size();
372 for (std::vector<CorrelatedGaussianObservables>::iterator it1 = CGO.begin(); it1 < CGO.end(); it1++)
373 obsbuffsize += it1->getObs().size();
374 buff[0] = new double[procnum * obsbuffsize];
375 for (int i = 1; i < procnum; i++) {
376 buff[i] = buff[i - 1] + obsbuffsize;
377 index_chain[i] = -1;
378 }
379
380 double ** sendbuff = new double *[procnum];
381 sendbuff[0] = new double[procnum * buffsize];
382 for (int il = 1; il < procnum; il++)
383 sendbuff[il] = sendbuff[il - 1] + buffsize;
384
385 while (mychain < fMCMCNChains) {
386 pars.clear();
387 pars = fMCMCStates.at(mychain).parameters;
388
389 if (PrintLoglikelihoodPlots || WriteParametersChain) {
390 std::map<std::string, double> tmpDPars;
391 setDParsFromParameters(pars, tmpDPars);
392 if (PrintLoglikelihoodPlots) DPars_allChains.push_back(tmpDPars);
393 if (WriteParametersChain) {
394 int k = 0;
395 for (std::map<std::string, double>::iterator it = tmpDPars.begin(); it != tmpDPars.end(); it++) hMCMCParameters[mychain][k++] = it->second;
396 }
397 }
398
399 index_chain[iproc] = mychain;
400 iproc++;
401 mychain++;
402 if (iproc < procnum && mychain < fMCMCNChains)
403 continue;
404
405 for (int il = 0; il < iproc; il++) {
406 //The first entry of the array specifies the task to be executed.
407
408 sendbuff[il][0] = 2.; // 2 = observables calculation
409 for (int im = 1; im < buffsize; im++) sendbuff[il][im] = fMCMCStates.at(index_chain[il]).parameters.at(im - 1);
410 }
411 for (int il = iproc; il < procnum; il++) {
412 sendbuff[il][0] = 3.; // 3 = nothing to execute, but return a buffer of observables
413 index_chain[il] = -1;
414 }
415 // double inittime = MPI::Wtime();
416 MPI_Scatter(sendbuff[0], buffsize, MPI_DOUBLE,
417 recvbuff, buffsize, MPI_DOUBLE,
418 0, MPI_COMM_WORLD);
419
420 if (recvbuff[0] == 2.) { // compute observables
421 double sbuff[obsbuffsize];
422 pars.assign(recvbuff + 1, recvbuff + buffsize);
423 setDParsFromParameters(pars,DPars);
424 Mod->Update(DPars);
425
426 int k = 0;
427 for (boost::ptr_vector<Observable>::iterator it = Obs_ALL.begin(); it < Obs_ALL.end(); it++) {
428 sbuff[k++] = it->computeTheoryValue();
429 }
430 for (std::vector<Observable2D>::iterator it = Obs2D_ALL.begin(); it < Obs2D_ALL.end(); it++) {
431 sbuff[k++] = it->computeTheoryValue();
432 sbuff[k++] = it->computeTheoryValue2();
433 }
434
435 for (std::vector<CorrelatedGaussianObservables>::iterator it1 = CGO.begin(); it1 < CGO.end(); it1++) {
436 std::vector<Observable> ObsV(it1->getObs());
437 for (std::vector<Observable>::iterator it = ObsV.begin(); it != ObsV.end(); ++it)
438 sbuff[k++] = it->computeTheoryValue();
439 }
440 MPI_Gather(sbuff, obsbuffsize, MPI_DOUBLE, buff[0], obsbuffsize, MPI_DOUBLE, 0, MPI_COMM_WORLD);
441 } else if (recvbuff[0] == 3.) { // do not compute observables, but gather the buffer
442 double sbuff[obsbuffsize];
443 MPI_Gather(sbuff, obsbuffsize, MPI_DOUBLE, buff[0], obsbuffsize, MPI_DOUBLE, 0, MPI_COMM_WORLD);
444 }
445
446 for (int il = 0; il < procnum; il++) {
447 if (index_chain[il] >= 0) {
448 int k = 0;
449 // fill the histograms for observables
450 int ko = 0, kweight = 0, k_all = 0, k_cObs = 0;
451 for (boost::ptr_vector<Observable>::iterator it = Obs_ALL.begin();
452 it < Obs_ALL.end(); it++) {
453 double th = buff[il][k++];
454 /* set the min and max of theory values */
455 if (th < thMin[it->getName()]) thMin[it->getName()] = th;
456 if (th > thMax[it->getName()]) thMax[it->getName()] = th;
457 Histo1D[it->getName()].GetHistogram()->Fill(th);
458 if (fMCMCFlagWriteChainToFile) hMCMCObservables[index_chain[il]][k_all++] = th;
459 else if (getchainedObsSize() > 0 && it->isWriteChain()) hMCMCObservables[index_chain[il]][k_cObs++] = th;
460 if (!it->isTMCMC()) {
461 obval[index_chain[il] * kmax + ko] = th;
462 ko++;
463 if (it->getDistr().compare("noweight") != 0 && it->getDistr().compare("writeChain") != 0) {
464 double weight = it->computeWeight(th);
465 obweight[index_chain[il] * kwmax + kweight] = weight;
466 if (fMCMCFlagWriteChainToFile) hMCMCObservables_weight[index_chain[il]][kweight] = weight;
467 kweight++;
468 }
469 }
470 }
471
472 // fill the 2D histograms for observables
473 for (std::vector<Observable2D>::iterator it = Obs2D_ALL.begin();
474 it < Obs2D_ALL.end(); it++) {
475 double th1 = buff[il][k++];
476 double th2 = buff[il][k++];
477 Histo2D[it->getName()].GetHistogram()->Fill(th1, th2);
478 if (fMCMCFlagWriteChainToFile) {
479 hMCMCObservables[index_chain[il]][k_all++] = th1;
480 hMCMCObservables[index_chain[il]][k_all++] = th2;
481 }
482 }
483
484 // fill the histograms for correlated observables
485 for (std::vector<CorrelatedGaussianObservables>::iterator it1 = CGO.begin(); it1 < CGO.end(); it1++) {
486 std::vector<Observable> ObsV(it1->getObs());
487 Double_t * COdata = new Double_t[ObsV.size()];
488 int nObs = 0;
489 for (std::vector<Observable>::iterator it = ObsV.begin(); it != ObsV.end(); ++it) {
490 double th = buff[il][k++];
491 /* set the min and max of theory values */
492 if (th < thMin[it->getName()]) thMin[it->getName()] = th;
493 if (th > thMax[it->getName()]) thMax[it->getName()] = th;
494 Histo1D[it->getName()].GetHistogram()->Fill(th);
495 if (fMCMCFlagWriteChainToFile) hMCMCObservables[index_chain[il]][k_all++] = th;
496 if (it1->isPrediction()) COdata[nObs++] = th;
497 }
498 if (it1->isPrediction()) CorrelationMap[it1->getName()]->AddRow(COdata);
499 delete [] COdata;
500 }
501 }
502 }
503 iproc = 0;
504 }
505 if (fMCMCFlagWriteChainToFile || getchainedObsSize() > 0 || WriteLogLikelihoodChain) InChainFillObservablesTree();
506 if (WriteParametersChain) InChainFillParametersTree();
507 delete sendbuff[0];
508 delete [] sendbuff;
509 delete [] recvbuff;
510 delete buff[0];
511 delete [] buff;
512#else
513 for (unsigned int i = 0; i < fMCMCNChains; ++i) {
514 // NOTE: BAT syncs fMCMCThreadLocalStorage with fMCMCStates before calling MCMCUserIterationInterface.
515 std::vector<double>::const_iterator first = fMCMCStates.at(i).parameters.begin();
516 std::vector<double>::const_iterator last = first + GetNParameters();
517 std::vector<double> currvec(first, last);
518 setDParsFromParameters(currvec,DPars);
519 if (PrintLoglikelihoodPlots) DPars_allChains.push_back(DPars);
520 if (WriteParametersChain) {
521 int k = 0;
522 for (std::map<std::string, double>::iterator it = DPars.begin(); it != DPars.end(); it++) hMCMCParameters[i][k++] = it->second;
523 }
524
525 Mod->Update(DPars);
526 // fill the histograms for observables
527 int k = 0, kweight = 0, k_all = 0, k_cObs = 0;
528 for (boost::ptr_vector<Observable>::iterator it = Obs_ALL.begin();
529 it < Obs_ALL.end(); it++) {
530 double th = it->computeTheoryValue();
531 /* set the min and max of theory values */
532 if (th < thMin[it->getName()]) thMin[it->getName()] = th;
533 if (th > thMax[it->getName()]) thMax[it->getName()] = th;
534 Histo1D[it->getName()].GetHistogram()->Fill(th);
535 if (fMCMCFlagWriteChainToFile) hMCMCObservables[i][k_all++] = th;
536 else if (getchainedObsSize() > 0 && it->isWriteChain()) hMCMCObservables[i][k_cObs++] = th;
537 if (!it->isTMCMC()) {
538 obval[i * kmax + k] = th;
539 k++;
540 if (it->getDistr().compare("noweight") != 0 && it->getDistr().compare("writeChain") != 0) {
541 double weight = it->computeWeight(th);
542 obweight[i * kwmax + kweight] = weight;
543 if (fMCMCFlagWriteChainToFile) hMCMCObservables_weight[i][kweight] = weight;
544 kweight++;
545 }
546 }
547 }
548
549 // fill the 2D histograms for observables
550 for (std::vector<Observable2D>::iterator it = Obs2D_ALL.begin();
551 it < Obs2D_ALL.end(); it++) {
552 double th1 = it->computeTheoryValue();
553 double th2 = it->computeTheoryValue2();
