86void NPSMEFTd6ATHDM::setNPSMEFTd6MFVParameters()
87{
88
89 //std::cout<<"Lambda_NP = "<<Lambda_NP<<std::endl;
90
91 //double Y1 = -getSMEFTCoeffEW("mh2") //check better the sign which depends on the definition in RGEsolver
92 //double Z1 = 2*getSMEFTCoeffEW("lambda")
93 // We do not need to include these two parameters since they are SM parameters. If we had to use them we'd just include
94 // these relations to extract them at the NP scale. However, the object getSMEFTCoeffEW takes the values from
95 // the object SMEFTEvolEW whose values are initialised and run up in NPSMEFTd6General::GenerateSMInitialConditions();
96 // WE WOULD NEED TO CALL THIS FUNCTION IN OUR postupdate as is done in the postupdate of NPSMEFTd6MFV if we'd like to
97 // access these parameters at this stage. There is no need of doing so since they do not enter in the tree-level matching.
98
99
100
101 CH_LNP = Z6*Z6;
102 //OuH[mif1_, mif2_] -> (Z6*lamU2bar[mif2, mif1])/MH2^2
103 CuH_0_LNP = Z6*etau;
104 CdH_0_LNP = Z6*etad;
105 CeH_0_LNP = Z6*etae;
106
107
108 // alphaOqu1[mif1_, mif2_, mif3_, mif4_] -> -1/6*(lamU2[mif3, mif2]*lamU2bar[mif4, mif1])/MH2^2
109 // The implemented term in the MFV model is Cqu1_y_LNP*YucL(mif1, mif4)*YuL(mif3, mif2)
110 // Check conventions carefully
111 Cqu1_y_LNP = -1/6*etau*etau;
112 Cqu8_y_LNP = -1*etau*etau;
113
114 Cqd1_y_LNP = -1/6*etad*etad;
115 Cqd8_y_LNP = -1*etad*etad;
116
117 //alphaOquqd1[mif1_, mif2_, mif3_, mif4_] -> (lamD2bar[mif4, mif3]*lamU2bar[mif2, mif1])/MH2^2
118 // The implemented term in the MFV model is Cquqd1_00_LNP*YdcL(mif3, mif4)*YuL(mif1, mif2)
119 Cquqd1_00_LNP = etad*etau;
120
121 //alphaOle[mif1_, mif2_, mif3_, mif4_] -> -((lamL2[mif3, mif2] lamL2bar[mif4, mif1])/(2 MH2^2))
122 Cle_y_LNP = (-0.5*etae*etae);
123
124 //alphaOledq[mif1_, mif2_, mif3_, mif4_] -> (lamD2[mif3, mif4] lamL2bar[mif2, mif1])/MH2^2
125 Cledq_00_LNP = (1.*etad*etae);
126
127 //alphaOlequ1[mif1_, mif2_, mif3_, mif4_] -> -((lamL2bar[mif2, mif1] lamU2bar[mif4, mif3])/MH2^2)
128 Clequ1_00_LNP = (-1.*etau*etae);
129
130
131}
