43 void
44 FlavorObservableError::setFromString( std::string const& valuesString )
45 {
46 std::string firstRemainder;
47 std::string secondRemainder;
48 minusUncertainty
49 = BOL::StringParser::stringToDouble( BOL::StringParser::firstWordOf(
50 valuesString,
51 &firstRemainder,
52 BOL::StringParser::whitespaceAndNewlineChars ) );
53 plusUncertainty
54 = BOL::StringParser::stringToDouble( BOL::StringParser::firstWordOf(
55 firstRemainder,
56 &secondRemainder,
57 BOL::StringParser::whitespaceAndNewlineChars ) );
58 evaluationScale
59 = BOL::StringParser::stringToDouble( BOL::StringParser::firstWordOf(
60 secondRemainder,
61 &firstRemainder,
62 BOL::StringParser::whitespaceAndNewlineChars ) );
63 int numberOfDaughterParticles( BOL::StringParser::stringToInt(
64 BOL::StringParser::firstWordOf(
65 firstRemainder,
66 &secondRemainder,
67 BOL::StringParser::whitespaceAndNewlineChars ) ) );
68 firstRemainder.assign( BOL::StringParser::trimFromFrontAndBack(
69 secondRemainder,
70 BOL::StringParser::whitespaceAndNewlineChars ) );
71 daughterParticleCodes.clear();
72 while( !(firstRemainder.empty()) )
73 {
74 daughterParticleCodes.push_back( BOL::StringParser::stringToInt(
75 BOL::StringParser::firstWordOf(
76 firstRemainder,
77 &secondRemainder,
78 BOL::StringParser::whitespaceAndNewlineChars ) ) );
79 firstRemainder.assign( BOL::StringParser::trimFromFrontAndBack(
80 secondRemainder,
81 BOL::StringParser::whitespaceAndNewlineChars ) );
82 }
83 if( ( 0 < numberOfDaughterParticles )
84 &&
85 ( (size_t)numberOfDaughterParticles != daughterParticleCodes.size() ) )
86 {
87 std::cout
88 << std::endl
89 << "LHPC::warning! An FOBSERR line declared a different number of"
90 << " daughter particles ( " << numberOfDaughterParticles << " ) to the"
91 << " actual number of daughter particle codes it had ( "
92 << daughterParticleCodes.size()
93 << " )! The declared number is being ignored in favor of the number of"
94 << " codes read in.";
95 std::cout << std::endl;
96 }
97 }

99 std::string
100 FlavorObservableError::getAsString() const
101 {
102 std::string
103 returnString( BOL::StringParser::doubleToString( minusUncertainty,
104 9,
105 3 ) );
106 returnString.append( FlavorObservable::spacesBetweenCodes,
107 ' ' );
108 returnString.append( BOL::StringParser::doubleToString( plusUncertainty,
109 9,
110 3 ) );
111 returnString.append( FlavorObservable::spacesBetweenCodes,
112 ' ' );
113 returnString.append( BOL::StringParser::doubleToString( evaluationScale,
114 9,
115 3 ) );
116 for( std::list< int >::const_iterator
117 daughterIterator( daughterParticleCodes.begin() );
118 daughterParticleCodes.end() != daughterIterator;
119 ++daughterIterator )
120 {
121 returnString.append( FlavorObservable::spacesBetweenCodes,
122 ' ' );
123 returnString.append( BOL::StringParser::intToSpacePaddedString(
124 *daughterIterator,
125 FlavorObservable::minimumDigitsForCodes,
126 "" ) );
127 }
128 return returnString;
129 }
