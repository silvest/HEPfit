103StandardModel* ModelFactory::CreateModel(const std::string& name)
104{
105 if (modelFactory.find(name) == modelFactory.end())
106 throw std::runtime_error("ERROR: Wrong model " + name + " passed to ModelFactory.\n");
107 return (modelFactory[name]());
108}
