77 double
78 ParticleLine::IPUP( int const whichComponent ) const
79 {
80 if( 1 == whichComponent )
81 {
82 return getXMomentum();
83 }
84 else if( 2 == whichComponent )
85 {
86 return getYMomentum();
87 }
88 else if( 3 == whichComponent )
89 {
90 return getZMomentum();
91 }
92 else if( 4 == whichComponent )
93 {
94 return getEnergy();
95 }
96 else if( 5 == whichComponent )
97 {
98 return getMass();
99 }
100 else
101 {
102 return BOL::UsefulStuff::notANumber;
103 }
104 }

106 bool
107 ParticleLine::recordLine( int const ownLineNumber,
108 std::string const& lineAsString )
109 /* this interprets a string as the data it is meant to correspond to. it
110 * returns false if the wrong number of data were given, & in that case
111 * it sets all its entries to BOL::UsefulStuff::reallyWrongValue. it
112 * returns true if all the entries were filled as expected. this function
113 * also clears daughterLines.
114 */
115 {
116 this->ownLineNumber = ownLineNumber;
117 daughterLines.clear();
118 primaryMotherLinePointer = NULL;
119 secondaryMotherLinePointer = NULL;
120 lineParser.clearEntries();
121 BOL::StringParser::parseByChar( lineAsString,
122 lineParser,
123 BOL::StringParser::whitespaceChars );
124 if( 13 != lineParser.getSize() )
125 // if the wrong number of data was given...
126 {
127 particleCode = (int)BOL::UsefulStuff::notANumber;
128 initialOrIntermediateOrFinalState = (int)BOL::UsefulStuff::notANumber;
129 primaryMotherLineNumber = (int)BOL::UsefulStuff::notANumber;
130 secondaryMotherLineNumber = (int)BOL::UsefulStuff::notANumber;
131 colorCode = (int)BOL::UsefulStuff::notANumber;
132 anticolorCode = (int)BOL::UsefulStuff::notANumber;
133 xMomentum = BOL::UsefulStuff::notANumber;
134 yMomentum = BOL::UsefulStuff::notANumber;
135 zMomentum = BOL::UsefulStuff::notANumber;
136 tMomentum = BOL::UsefulStuff::notANumber;
137 restMass = BOL::UsefulStuff::notANumber;
138 properLifetime = BOL::UsefulStuff::notANumber;
139 twiceSpin = (int)BOL::UsefulStuff::notANumber;
140 return false;
141 }
142 else
143 {
144 particleCode = BOL::StringParser::stringToInt( lineParser[ 0 ] );
145 initialOrIntermediateOrFinalState
146 = BOL::StringParser::stringToInt( lineParser[ 1 ] );
147 primaryMotherLineNumber
148 = BOL::StringParser::stringToInt( lineParser[ 2 ] );
149 secondaryMotherLineNumber
150 = BOL::StringParser::stringToInt( lineParser[ 3 ] );
151 colorCode = BOL::StringParser::stringToInt( lineParser[ 4 ] );
152 anticolorCode = BOL::StringParser::stringToInt( lineParser[ 5 ] );
153 xMomentum = BOL::StringParser::stringToDouble( lineParser[ 6 ] );
154 yMomentum = BOL::StringParser::stringToDouble( lineParser[ 7 ] );
155 zMomentum = BOL::StringParser::stringToDouble( lineParser[ 8 ] );
156 tMomentum = BOL::StringParser::stringToDouble( lineParser[ 9 ] );
157 restMass = BOL::StringParser::stringToDouble( lineParser[ 10 ] );
158 properLifetime = BOL::StringParser::stringToDouble( lineParser[ 11 ] );
159 twiceSpin = BOL::StringParser::stringToInt( lineParser[ 12 ] );
160 return true;
161 }
162 }
