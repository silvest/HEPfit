21 SlhaValueColoredSegments::SlhaValueColoredSegments(
22 std::string const& colorDefinitionXml,
23 double const coloredLineWidth,
24 LHPC::SlhaSimplisticInterpreter& slhaParser ) :
25 SlhaValueLineColoring(),
26 colorsWithWidths(),
27 cumulativeWeightTotal( 0.0 )
28 {
29 BOL::AsciiXmlParser xmlParser;
30 if( xmlParser.loadString( colorDefinitionXml ) )
31 {
32 while( xmlParser.readNextElement() )
33 {
34 if( xmlParser.currentElementNameMatches( "LineSegment" ) )
35 {
36 addSegment( xmlParser.getTrimmedCurrentElementContent(),
37 slhaParser );
38 }
39 }
40
41 // Now that all the weights have been read, their widths can be set.
42 if( cumulativeWeightTotal == 0.0 )
43 {
44 double const equalWeight( coloredLineWidth
45 / static_cast< double >( colorsWithWidths.size() ) );
46 for( std::vector< std::pair< std::string, double > >::iterator
47 colorWithWidth( colorsWithWidths.begin() );
48 colorWithWidth < colorsWithWidths.end();
49 ++colorWithWidth )
50 {
51 colorWithWidth->second = equalWeight;
52 }
53 }
54 else
55 {
56 double const
57 normalizationFactor( coloredLineWidth / cumulativeWeightTotal );
58
59 for( std::vector< std::pair< std::string, double > >::iterator
60 colorWithWidth( colorsWithWidths.begin() );
61 colorWithWidth < colorsWithWidths.end();
62 ++colorWithWidth )
63 {
64 colorWithWidth->second *= normalizationFactor;
65 }
66 }
67 }
68 else
69 {
70 throw std::runtime_error( "Could not parse <LineColor>." );
71 }
72 }

80 // This adds a color with an unnormalized line segment length weight to
81 // colorsWithWidths based on the XML in combination with the given SLHA
82 // values, as well as updating the weight total. The widths of the segments
83 // will be correctly normalized later in the constructor.
84 void SlhaValueColoredSegments::addSegment( std::string segmentXml,
85 LHPC::SlhaSimplisticInterpreter& slhaParser )
86 {
87 BOL::AsciiXmlParser xmlParser;
88 if( xmlParser.loadString( segmentXml ) )
89 {
90 std::string segmentColor( "" );
91 std::string segmentValue( "0.0" );
92 double powerOfValueForWeight( 1.0 );
93 while( xmlParser.readNextElement() )
94 {
95 if( xmlParser.currentElementNameMatches( "SegmentColor" ) )
96 {
97 segmentColor.assign( xmlParser.getTrimmedCurrentElementContent() );
98 }
99 else if( xmlParser.currentElementNameMatches( "ColorWeight" ) )
100 {
101 std::map< std::string, std::string > const&
102 weightAttributes( xmlParser.getCurrentElementAttributes() );
103 std::map< std::string, std::string >::const_iterator
104 attributeFinder( weightAttributes.find("RaiseToPower") );
105 if( attributeFinder != weightAttributes.end() )
106 {
107 powerOfValueForWeight
108 = BOL::StringParser::stringToDouble( attributeFinder->second );
109 }
110 segmentValue.assign( xmlParser.getTrimmedCurrentElementContent() );
111 }
112 }
113 double interpretedValue( 0.0 );
114
115 // The evaluation of the conditional will set interpretedValue
116 // correctly if it is just a number (i.e. if stringIsDouble returns
117 // true).
118 if( !(BOL::StringParser::stringIsDouble( segmentValue,
119 interpretedValue )) )
120 {
121 interpretedValue = slhaParser.getDouble( segmentValue );
122 }
123
124 // Now that the value has been correctly parsed, we can turn it into a
125 // weight.
126 double segmentWeight = pow( interpretedValue, powerOfValueForWeight );
127 cumulativeWeightTotal += segmentWeight;
128 colorsWithWidths.push_back( std::make_pair( segmentColor,
129 segmentWeight ) );
130 }
131 else
132 {
133 throw std::runtime_error( "Could not parse <LineSegment>." );
134 }
135 }
