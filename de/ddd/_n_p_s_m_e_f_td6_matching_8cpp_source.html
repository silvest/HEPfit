24void NPSMEFTd6Matching::updateNPSMEFTd6Parameters()
25{
26
27 SMEFT_factor = (M_PI/myNPSMEFTd6.getAle())*(myNPSMEFTd6.v()/myNPSMEFTd6.getLambda_NP())*(myNPSMEFTd6.v()/myNPSMEFTd6.getLambda_NP())/myNPSMEFTd6.getCKM().computelamt_s();
28 Muw = myNPSMEFTd6.getMuw();
29
30 double ytop = myNPSMEFTd6.getQuarks(QCD::TOP).getMass()/myNPSMEFTd6.v();
31 loop_factor = myNPSMEFTd6.getCKM().computelamt_s()*ytop*ytop/(16.*M_PI*M_PI);
32
33 C7NP = 0.; // to be implemented
34 C8NP = 0.; // to be implemented
35 C7pNP = 0.; // to be implemented
36 C8pNP = 0.; // to be implemented
37
38 double logLambdaomu = log(myNPSMEFTd6.getLambda_NP()/Muw);
39
40 CLQ1_1123_tot = myNPSMEFTd6.getCLQ1_1123();
41 CLQ1_1123_tot -= loop_factor*logLambdaomu*(myNPSMEFTd6.getCHL1_11()-myNPSMEFTd6.getCLu_1133());
42 CLQ1_2223_tot = myNPSMEFTd6.getCLQ1_2223();
43 CLQ1_2223_tot -= loop_factor*logLambdaomu*(myNPSMEFTd6.getCHL1_22()-myNPSMEFTd6.getCLu_2233());
44 CLQ3_1123_tot = myNPSMEFTd6.getCLQ3_1123();
45 CLQ3_1123_tot += loop_factor*logLambdaomu*myNPSMEFTd6.getCHL3_11();
46 CLQ3_2223_tot = myNPSMEFTd6.getCLQ3_2223();
47 CLQ3_2223_tot += loop_factor*logLambdaomu*myNPSMEFTd6.getCHL3_22();
48 CQe_2311_tot = myNPSMEFTd6.getCQe_2311();
49 CQe_2311_tot -= loop_factor*logLambdaomu*(myNPSMEFTd6.getCHe_11()-myNPSMEFTd6.getCeu_1133());
50 CQe_2322_tot = myNPSMEFTd6.getCQe_2322();
51 CQe_2322_tot -= loop_factor*logLambdaomu*(myNPSMEFTd6.getCHe_22()-myNPSMEFTd6.getCeu_2233());
52
53 C9NPmu = SMEFT_factor*(CQe_2322_tot+CLQ1_2223_tot+CLQ3_2223_tot);
54 C10NPmu = SMEFT_factor*(CQe_2322_tot-CLQ1_2223_tot-CLQ3_2223_tot);
55 C9pNPmu = SMEFT_factor*(myNPSMEFTd6.getCed_2223()+myNPSMEFTd6.getCLd_2223());
56 C10pNPmu = SMEFT_factor*(myNPSMEFTd6.getCed_2223()-myNPSMEFTd6.getCLd_2223());
57 CSNPmu = SMEFT_factor*myNPSMEFTd6.getCLedQ_22();
58 CSpNPmu = SMEFT_factor*myNPSMEFTd6.getCpLedQ_22();
59 CPNPmu = -SMEFT_factor*CSNPmu;
60 CPpNPmu = SMEFT_factor*CSpNPmu;
61
62 C9NPe = SMEFT_factor*(CQe_2311_tot+CLQ1_1123_tot+CLQ3_1123_tot);
63 C10NPe = SMEFT_factor*(CQe_2311_tot-CLQ1_1123_tot-CLQ3_1123_tot);
64 C9pNPe = SMEFT_factor*(myNPSMEFTd6.getCed_1123()+myNPSMEFTd6.getCLd_1123());
65 C10pNPe = SMEFT_factor*(myNPSMEFTd6.getCed_1123()-myNPSMEFTd6.getCLd_1123());
66 CSNPe = SMEFT_factor*myNPSMEFTd6.getCLedQ_11();
67 CSpNPe = SMEFT_factor*myNPSMEFTd6.getCpLedQ_11();
68 CPNPe = -SMEFT_factor*CSNPe;
69 CPpNPe = SMEFT_factor*CSpNPe;
70
71 StandardModelMatching::updateSMParameters();
72}
