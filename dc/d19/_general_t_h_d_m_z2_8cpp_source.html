69void GeneralTHDMZ2::setParameter(const std::string name, const double& value)
70{
71 if (name.compare("tanb") == 0) {
72 tanb = value;
73 if(tanb > 0.) {
74 beta = atan(tanb);
75 cosb = cos(beta);
76 cos2b = cos(2.*beta);
77 cos4b = cos(4.*beta);
78 cos6b = cos(6.*beta);
79 sinb = sin(beta);
80 sin2b = sin(2.*beta);
81 sin4b = sin(4.*beta);
82 sin6b = sin(6.*beta);
83 }
84 else {
85 throw std::runtime_error("ERROR: in GeneralTHDMZ2::setParameter, tanb < 0!");
86 }
87 }
88 else if (name.compare("bma") == 0) {
89 bma = value;
90 cos2bma = cos(2.*bma);
91 sin2bma = sin(2.*bma);
92 }
93 else if (name.compare("m12_2") == 0) {
94 m12_2 = value;
95 M2aux = m12_2/cosb/sinb;
96 }
97 else if (name.compare("alpha1") == 0)
98 GeneralTHDM::setParameter("alpha1", bma - M_PI/2.);
99 else if (name.compare("lambda2") == 0) {
100 mh_2 = getmH1sq();
101 mH_2 = getmH2sq();
102 mA_2 = getmH3sq();
103 mHp_2 = getmHp2();
104 vev = GeneralTHDM::v();
105 GeneralTHDM::setParameter("lambda2", ((4.*(mh_2 - mH_2)*(3. + 5.*cos4b)*cos2bma/cosb/sinb +
106 (-32.*m12_2*(1. + cos4b) + (mh_2 + mH_2)*(7.*sin2b + 3.*sin6b))/cosb/cosb/sinb/sinb +
107 (mH_2 - mh_2)*(20. - cos2b + 12.*cos4b + cos6b)*2.*sin2bma)/64./cosb/sinb/vev/vev));
108 }
109 else if (name.compare("lambda3") == 0)
110 GeneralTHDM::setParameter("lambda3", ((-32.*m12_2 + 8.*(mh_2 + mH_2 + mA_2)*sin2b +
111 (mh_2 - mH_2)*(-4. + 3.*cos2b + 4.*cos4b + cos6b)*sin2bma)/16./cosb/sinb/vev/vev));
112 else if (name.compare("Relambda7") == 0)
113 GeneralTHDM::setParameter("Relambda7", ((4.*(mh_2 - mH_2)*cos2b*cos2bma/cosb/sinb +
114 4.*(mh_2 + mH_2 - 2.*m12_2/cosb/sinb)*sin4b/sin2b/sin2b -
115 (mh_2 - mH_2)*(1. + 8.*cos2b + cos4b)*sin2bma)/8./vev/vev));
116 else if (name.compare("Nu_11r") == 0) {
117 if (flag_model == "type1")
118 GeneralTHDM::setParameter("Nu_11r", 1./tanb);
119 else if (flag_model == "type2")
120 GeneralTHDM::setParameter("Nu_11r", 1./tanb);
121 else if (flag_model == "typeX")
122 GeneralTHDM::setParameter("Nu_11r", 1./tanb);
123 else if (flag_model == "typeY")
124 GeneralTHDM::setParameter("Nu_11r", 1./tanb);
125 }
126 else if (name.compare("Nd_11r") == 0) {
127 if (flag_model == "type1")
128 GeneralTHDM::setParameter("Nd_11r", 1./tanb);
129 else if (flag_model == "type2")
130 GeneralTHDM::setParameter("Nd_11r", -tanb);
131 else if (flag_model == "typeX")
132 GeneralTHDM::setParameter("Nd_11r", 1./tanb);
133 else if (flag_model == "typeY")
134 GeneralTHDM::setParameter("Nd_11r", -tanb);
135 }
136 else if (name.compare("Nl_11r") == 0) {
137 if (flag_model == "type1")
138 GeneralTHDM::setParameter("Nl_11r", 1./tanb);
139 else if (flag_model == "type2")
140 GeneralTHDM::setParameter("Nl_11r", -tanb);
141 else if (flag_model == "typeX")
142 GeneralTHDM::setParameter("Nl_11r", -tanb);
143 else if (flag_model == "typeY")
144 GeneralTHDM::setParameter("Nl_11r", 1./tanb);
145 }
146 else
147 GeneralTHDM::setParameter(name, value);
148}

150bool GeneralTHDMZ2::CheckParameters(const std::map<std::string, double>& DPars)
151{
152 for (int i = 0; i < NGeneralTHDMZ2vars; i++) {
153 if (DPars.find(GeneralTHDMZ2vars[i]) == DPars.end()) {
154 std::cout << "ERROR: missing mandatory GeneralTHDMZ2 parameter " << GeneralTHDMZ2vars[i] << std::endl;
155 raiseMissingModelParameterCount();
156 addMissingModelParameter(GeneralTHDMZ2vars[i]);
157 }
158 }
159 return(GeneralTHDM::CheckParameters(DPars));
160}
