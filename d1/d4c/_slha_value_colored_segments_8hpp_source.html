68 // This writes the sets of colors as segments with lengths given by their
69 // fractional weighting of width of the line for the value given to the
70 // constructor.
71 inline void SlhaValueColoredSegments::writeLineData( int& gnuplotLineIndex,
72 std::ofstream& gnuplotDataFile,
73 std::ofstream& gnuplotCommandFile,
74 double const leftEndHorizontalPosition,
75 double const lineVerticalPositon,
76 double const rightEndHorizontalPosition ) const
77 {
78 double segmentLeftEnd( leftEndHorizontalPosition );
79 for( std::vector< std::pair< std::string, double > >::const_iterator
80 colorWithWidth( colorsWithWidths.begin() );
81 colorWithWidth < colorsWithWidths.end();
82 ++colorWithWidth )
83 {
84 double segmentRightEnd( segmentLeftEnd + colorWithWidth->second );
85 AddLine( gnuplotLineIndex,
86 gnuplotDataFile,
87 gnuplotCommandFile,
88 segmentLeftEnd,
89 lineVerticalPositon,
90 segmentRightEnd,
91 lineVerticalPositon,
92 colorWithWidth->first );
93 segmentLeftEnd = segmentRightEnd;
94 }
