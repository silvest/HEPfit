56 // neutralScalarsAndPseudoscalarPointers should start with the scalar
57 // from a StandardModel instance.
58 MassEigenstate* vectorFiller( MassEigenstate::findPointerWithCode(
59 PDGIX::neutralColorlessScalarOne,
60 pdgCodeMap ) );
61 if( NULL == vectorFiller )
62 {
63 vectorFiller = MassEigenstate::findPointerWithCode(
64 PDGVII::neutralColorlessScalarOne,
65 pdgCodeMap );
66 }
67 if( NULL != vectorFiller )
68 {
69 neutralScalarsAndPseudoscalarPointers.push_back( vectorFiller );
70 }
71 neutralScalarsAndPseudoscalarPointers.push_back(
72 &neutralColorlessScalarTwo );
73 neutralScalarsAndPseudoscalarPointers.push_back(
74 &neutralColorlessPseudoscalarOne );
75 chargedColorlessSpinZeroBosonPointers[ 1 ]
76 = &negativeColorlessSpinZeroBosonOne;
77 chargedEwsbSpinZeroAndOneBosonPointers[ 1 ]
78 = chargedColorlessSpinZeroBosonPointers[ 1 ];
79 vectorFiller = MassEigenstate::findPointerWithCode( PDGIX::wPlusBosonOne,
80 pdgCodeMap );
81 if( NULL == vectorFiller )
82 {
83 vectorFiller
84 = MassEigenstate::findPointerWithCode( PDGVII::wPlusBosonOne,
85 pdgCodeMap );
86 }
87 if( NULL != vectorFiller )
88 {
89 chargedEwsbSpinZeroAndOneBosonPointers.push_back( vectorFiller );
90 chargedEwsbSpinZeroAndOneBosonPointers.push_back(
91 &(vectorFiller->getChargeConjugate()) );
92 }
93 neutralEwsbSpinZeroAndOneBosonPointers
94 = neutralScalarsAndPseudoscalarPointers;
95 vectorFiller = MassEigenstate::findPointerWithCode( PDGIX::zBosonOne,
96 pdgCodeMap );
97 if( NULL == vectorFiller )
98 {
99 vectorFiller
100 = MassEigenstate::findPointerWithCode( PDGVII::zBosonOne,
101 pdgCodeMap );
102 }
103 if( NULL != vectorFiller )
104 {
105 neutralEwsbSpinZeroAndOneBosonPointers.push_back( vectorFiller );
106 }
107 ewsbSpinZeroAndOneBosonPointers = neutralEwsbSpinZeroAndOneBosonPointers;
108 ewsbSpinZeroAndOneBosonPointers.insert(
109 ewsbSpinZeroAndOneBosonPointers.end(),
110 chargedEwsbSpinZeroAndOneBosonPointers.begin(),
111 chargedEwsbSpinZeroAndOneBosonPointers.end() );
112 }
