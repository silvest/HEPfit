42 bool
43 LhefParser::readNextEvent()
44 // this reads in the next event in the event file, & returns true if
45 // successful.
46 {
47 if( fileIsOpen )
48 {
49 fileIsOpen = fileParser.readNextElement();
50 while( !(fileParser.currentElementNameMatches( eventTag )) )
51 {
52 fileIsOpen = fileParser.readNextElement();
53 if( !fileIsOpen )
54 {
55 fileParser.closeFile();
56 eventAsString.assign( "" );
57 return false;
58 }
59 }
60 eventAsString.assign( fileParser.getCurrentElementContent() );
61 eventIsValid = currentEvent.recordEvent( eventAsString );
62 for( int filterIndex( automaticFilters.size() - 1 );
63 0 <= filterIndex;
64 --filterIndex )
65 {
66 automaticFilters[ filterIndex ]->updateForNewEvent( currentEvent );
67 }
68 }
69 return fileIsOpen;
70 }
