65const double NPSTUVWXY::epsilon1() const
66{
67 double c2 = trueSM.cW2();
68 double s2 = trueSM.sW2();
69 double eps1 = trueSM.epsilon1();
70 eps1 += myObliqueThat - myObliqueW + 2.0 * sqrt(s2) / sqrt(c2) * myObliqueX
71 - s2 / c2*myObliqueY;
72 return eps1;
73}

102const double NPSTUVWXY::obliqueS() const
103{
104 double sW2_SM = trueSM.sW2();
105 double sW_SM = sqrt(sW2_SM);
106 double cW_SM = trueSM.cW2();
107 return ( (myObliqueShat - myObliqueW + myObliqueX / (sW_SM * cW_SM) - myObliqueY)
108 * 4.0 * sW2_SM / alphaMz());
109}
110
111const double NPSTUVWXY::obliqueT() const
112{
113 double sW2_SM = trueSM.sW2();
114 double sW_SM = sqrt(sW2_SM);
115 double cW2_SM = trueSM.cW2();
116 double cW_SM = sqrt(cW2_SM);
117 return ( (myObliqueThat - myObliqueW + 2.0 * sW_SM / cW_SM * myObliqueX
118 - sW2_SM / cW2_SM * myObliqueY) / alphaMz());
119}
120
121const double NPSTUVWXY::obliqueU() const
122{
123 double sW2_SM = trueSM.sW2();
124 double sW_SM = sqrt(sW2_SM);
125 double cW_SM = sqrt(trueSM.cW2());
126 return ( (-myObliqueUhat + myObliqueV + myObliqueW
127 - 2.0 * sW_SM / cW_SM * myObliqueX)*4.0 * sW2_SM / alphaMz());
128}

133const double NPSTUVWXY::GammaW() const
134{
135 double Gamma_W = trueSM.GammaW();
136
137 double Wbar = (obliqueV() - obliqueW()) / alphaMz();
138
139 double alpha = StandardModel::alphaMz();
140 double c2 = trueSM.cW2();
141 double s2 = trueSM.sW2();
142
143 Gamma_W *= 1.0 - 3.0 * alpha / 4.0 / (c2 - s2)
144 *(obliqueS() - 2.0 * c2 * obliqueT()
145 - (c2 - s2) * obliqueU() / 2.0 / s2 - 2.0 * (c2 - s2) * Wbar)
146 - (1.0 + c2) / 2.0 / (c2 - s2) * DeltaGF();
147
148 return Gamma_W;
149}
