20double unitarity00eveP_Z2::computeThValue()
21{
22 gslpp::complex i = gslpp::complex::i();
23
24 double la1Q = myGTHDM.getlambda1_Z2();
25 double la2Q = myGTHDM.getlambda2_Z2();
26 double la3Q = myGTHDM.getlambda3_Z2();
27 double la4Q = myGTHDM.getlambda4_Z2();
28 double la5Q = myGTHDM.getlambda5_Z2();
29
30 double beta1 = myGTHDM.getMyGTHDMCache()->betalambda1_Z2(la1Q, la3Q, la4Q, la5Q);
31 double beta2 = myGTHDM.getMyGTHDMCache()->betalambda2_Z2(la2Q, la3Q, la4Q, la5Q);
32 double beta3 = myGTHDM.getMyGTHDMCache()->betalambda3_Z2(la1Q, la2Q, la3Q, la4Q, la5Q);
33 double beta4 = myGTHDM.getMyGTHDMCache()->betalambda4_Z2(la1Q, la2Q, la3Q, la4Q, la5Q);
34
35 gslpp::complex B1 = -3.*la1Q + 9.*beta1/2. + (i*M_PI - 1.)*(9.*la1Q*la1Q +
36 (2.*la3Q + la4Q)*(2.*la3Q + la4Q))/16./M_PI/M_PI;
37
38 gslpp::complex B2 = -3.*la2Q + 9.*beta2/2. + (i*M_PI - 1.)*(9.*la2Q*la2Q +
39 (2.*la3Q + la4Q)*(2.*la3Q + la4Q))/16./M_PI/M_PI;
40
41 gslpp::complex B3 = -2.*la3Q - la4Q + 3.*(2.*beta3 + beta4)/2. + 3.*(i*M_PI -
42 1.)*(la1Q + la2Q)*(2.*la3Q + la4Q)/16./M_PI/M_PI;
43
44 return ((B1 + B2 + sqrt((B1 - B2)*(B1 - B2) + 4.*B3*B3))/32./M_PI - i/2.).abs();
45}
