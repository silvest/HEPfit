68 inline void SlhaValueLineColoring::AddLine( int& gnuplotLineIndex,
69 std::ofstream& gnuplotDataFile,
70 std::ofstream& gnuplotCommandFile,
71 double const leftEndHorizontalPosition,
72 double const leftEndVerticalPosition,
73 double const rightEndHorizontalPosition,
74 double const rightEndVerticalPosition,
75 std::string const& lineColor )
76 {
77 if( lineColor.compare( "transparent" ) != 0 )
78 {
79 gnuplotDataFile
80 << leftEndHorizontalPosition << " " << leftEndVerticalPosition
81 << std::endl
82 << rightEndHorizontalPosition << " " << rightEndVerticalPosition
83 << std::endl;
84 // Double blank lines are treated by gnuplot as meaning that the data
85 // above has a separate "index":
86 gnuplotDataFile << std::endl << std::endl;
87 gnuplotCommandFile
88 << "set style line " << (++gnuplotLineIndex) << " lt rgb \""
89 << lineColor << "\" lw 3"
90 << std::endl;
91 }
