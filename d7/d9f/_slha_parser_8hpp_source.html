162 inline void
163 SlhaParser::clearBlocks()
164 // this goes through all the blocks in blockMap & calls their clearEntries()
165 // member functions.
166 {
167 /* observingSpectrumUpdater is set so that only decays will be updated
168 * in the observing spectra until later, when the mass map pointers will be
169 * set so that the masses can be recorded.
170 */
171 observingSpectrumUpdater.setFmassMap( NULL );
172 observingSpectrumUpdater.setMassMap( NULL );
173 updateObservers();
174 // the MassSpectrum class over-rides respondToObservedSignal() to call its
175 // clearMassesAndDecays() function.
176 blockMapIterator = blockMap.begin();
177 while( blockMap.end() != blockMapIterator )
178 {
179 blockMapIterator->second->clearEntries();
180 ++blockMapIterator;
181 }
182 currentBlockPointer = NULL;
183 }

185 inline void
186 SlhaParser::prepareToReadNewDecay()
187 // this parses the block header line & sets currentMassEigenstate
188 // appropriately, & records its decay width.
189 {
190 prepareForEitherBlockOrDecay();
191 if( !(observerList.empty())
192 &&
193 ( 3 == wordsOfLine.getSize() ) )
194 // if there is a spectrum for recording decays, & if the line has the
195 // right number of entries ("DECAY", particle code, decay width)...
196 {
197 observingSpectrumUpdater.recordDecayHeader( wordsOfLine[ 1 ],
198 wordsOfLine[ 2 ] );
199 }
200 }

202 inline void
203 SlhaParser::recordDecayLine()
204 // this interprets the current line as a decay for the spectrum.
205 {
206 wordsOfLine.clearEntries();
207 BOL::StringParser::parseByChar( dataString,
208 wordsOfLine,
209 BOL::StringParser::whitespaceChars );
210 if( !(wordsOfLine.isEmpty()) )
211 // if there is a decay to record...
212 {
213 observingSpectrumUpdater.recordDecayLine( wordsOfLine );
214 }
215 }
216
217 inline void
218 SlhaParser::finishUpEitherBlockOrDecay()
219 // this either pushes a read decay to the observing spectra or gets the
220 // read block to be pushed to its observers.
221 {
222 if( NULL != currentBlockPointer )
223 {
224 currentBlockPointer->finishRecordingLines();
225 currentBlockPointer = NULL;
226 }
227 if( observingSpectrumUpdater.isHoldingDecay() )
228 {
229 updateObservers( observingSpectrumUpdater );
230 observingSpectrumUpdater.clearDecay();
231 }
232 }
233
234 inline void
235 SlhaParser::ensureSpectraRecordMasses()
236 // this reads the masses from the FMASS & MASS blocks into the spectrum, if
237 // necessary.
238 {
239 if( !(observerList.empty()) )
240 {
241 observingSpectrumUpdater.setFmassMap( fmassBlockPointer->getFmassMap() );
242 observingSpectrumUpdater.setFmasserrMap(
243 fmasserrBlockPointer->getFmasserrMap() );
244 observingSpectrumUpdater.setMassMap( massBlockPointer->getMassMap() );
245 updateObservers( observingSpectrumUpdater );
246 }
247 }
248
249 inline void
250 SlhaParser::prepareForEitherBlockOrDecay()
251 // this sets up the common parsing of the line being read.
252 {
253 finishUpEitherBlockOrDecay();
254 wordsOfLine.clearEntries();
255 BOL::StringParser::parseByChar( dataString,
256 wordsOfLine,
257 BOL::StringParser::whitespaceChars );
258 }
