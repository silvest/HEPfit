215 lambdaHUV = getSMEFTCoeffEW("lambda");
216
217 g1UV2 = g1UV * g1UV;
218 g2UV2 = g2UV * g2UV;
219 g3UV2 = g3UV * g3UV;
220
221 for (int i = 0; i < 3; i++) {
222 for (int j = 0; j < 3; j++) {
223 YuUV.assignre(i, j, getSMEFTCoeffEW("YuR", i, j) );
224 YuUV.assignim(i, j, getSMEFTCoeffEW("YuI", i, j) );
