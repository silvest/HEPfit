33bool Flavour::setFlag(const std::string name, const bool value)
34{
35
36 if (name.compare("UseDispersionRelation") == 0) {
37 setFlagUseDispersionRelation(value);
38 return true;
39 } else if (name.compare("UsezExpansion") == 0) {
40 setFlagUsezExpansion(value);
41 return true;
42 } else if (name.compare("CLNflag") == 0) {
43 setFlagCLN(value);
44 return true;
45 } else if (name.compare("BGLflag") == 0) {
46 setFlagBGL(value);
47 return true;
48 } else if (name.compare("DMflag") == 0) {
49 setFlagDM(value);
50 return true;
51 } else if (name.compare("FixedWCbtos") == 0) {
52 setFlagFixedWCbtos(value);
53 return true;
54 } else
55 return false;
56}

279{
280 std::vector<int> key({args...});

283 }
284 return *map.at(key);
